get_mean_abs_corr <- function(res.cs, dat_indx){
  res.mac <- sapply(dat_indx, function(i) {
  if (!is.null(res.cs[[i]])) {
    return(res.cs[[i]]$purity$mean.abs.corr)
  } else {
    return(NULL)
  }
})
  return(unlist(res.mac))
}
